# gf-ui

一个基于 React 开发的组件库，包含基础组件、业务组件、区块、hooks、utils 等
npm i gf-react-ui -D

[
'feat', // 新功能（feature）
'improve', // 改进功能
'fix', // 修补 bug
'docs', // 文档（documentation）
'style', // 格式（不影响代码运行的变动）
'refactor', // 重构（即不是新增功能，也不是修改 bug 的代码变动）
'test', // 增加测试
'revert', // 回滚
'config', // 构建过程或辅助工具的变动
'chore', // 其他改动
